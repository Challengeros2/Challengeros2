def zad1a(tekst):
    licznik = {}
    for znak in tekst:
        if znak in licznik:
            licznik[znak] += 1
        else:
            licznik[znak] = 1
    return licznik

def zad1d(napis):
    napis = napis.lower()
    licznik = {}

    for znak in napis:
        if znak in licznik:
            licznik[znak] += 1
        else:
            licznik[znak] = 1

    return max(licznik, key=licznik.get)

def zad2():
    wystapienia = {}

    while True:
        dane = input("Podaj liczbę całkowitą (Enter kończy): ")

        if dane == "":
            break

        try:
            liczba = int(dane)
            if liczba in wystapienia:
                wystapienia[liczba] += 1
            else:
                wystapienia[liczba] = 1
        except ValueError:
            print("Błąd: podana wartość nie jest liczbą całkowitą.")

    print("Słownik wystąpień:", wystapienia)
